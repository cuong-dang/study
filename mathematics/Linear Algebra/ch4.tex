\subsection{Introduction to Eigenvalues and Eigenvectors}
\begin{definition*}
	Let $A$ be an $n\times n$ matrix. A scalar $\lambda$ is called an eigenvalue of $A$ if there is a nonzero
	vector $\vx$ such that $A\vx=\lambda\vx$. Such a vector $\vx$ is called an eigenvector of $A$
	corresponding to $\lambda$.
\end{definition*}
