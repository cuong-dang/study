\subsection{The Precise Definition of a Limit}
\subsubsection*{Example 4} Prove that $\lim_{x\to3}x^2 = 9$.
\begin{proof}
  Given $\epsilon > 0$, let $\delta = \min\left\{1, \epsilon / 7\right\}$. If $0 < |x-3| <
    \delta$, then $|x-3| < 1 \implies -1 < x-3 < 1 \implies 5 < x+3 < 7$, or $|x+3| < 7$.
  We also have $|x-3| < \epsilon/7$, so
  \[\left|x^2-9\right| = |x-3||x+3| < \frac{\epsilon}{7}\cdot7 = \epsilon.\]
  This shows that $\lim_{x\to3}x^2 = 9$.
\end{proof}

\subsection{Continuity}
\begin{theorem*}[The Intermediate Value Theorem]
  Suppose that $f$ is continuous on the closed interval $[a, b]$ and let $N$ be any number
  between $f(a)$ and $f(b)$, where $f(a) \neq f(b)$. Then there exists a number $c$ in $(a, b)$
  such that $f(c) = N$.
  \begin{proof}
    Let set $H = \left\{x\in[a,b]:f(x)<N\right\} \neq \emptyset$. Then, $\exists c =
      \sup(H)$. We will now show that $f(c) = N$ by contradiction. Suppose $f(c) < N$.
    Then $N-f(c)>0$. Let $\epsilon=N-f(c)>0$. $f$ is continuous $\implies \lim_{x \to c} =
      f(c)\implies\exists\delta: 0<|x-c|<\delta\implies|f(x)-f(c)|<\epsilon=N-f(c)$.
    Choose $x=c+\frac{\delta}{2}$. Then, $f(x)-f(c)<N-f(c)\implies f(x)<N\implies c+
      \frac{\delta}{2}\in H$. This contradicts the fact that $c=\sup(H)$. Suppose $f(c)
      > N$. Then $f(c)-N>0$. Let $\varepsilon=f(c)-N>0$. Again, $f$ is continuous
    $\implies\lim_{x \to c}=f(c)\implies\exists\delta: 0<|x-c|<\delta\implies|f(x)-f(c)|<
      \varepsilon=f(c)-N$. Choose $x=c-\frac{\delta}{2}$. Then, $N-f(c)<f(x)-f(c)
      \implies f(x)>N\implies c-\frac{\delta}{2}>N$. Then $c$ cannot be $\sup(H)$, a
    contradiction. Therefore, $f(c) = N$.
  \end{proof}
\end{theorem*}

\begin{proposition*}
  Suppose $f$ and $g$ are functions. If $\lim_{t \to b}f(t)=L$, $\lim_{x \to a}g(x)=b$, and $g(x)
    \neq b$ for all x sufficiently close to $a$, then $\lim_{x \to a}f\left(g(x)\right)=L$.
\end{proposition*}
\begin{proof}
  $\lim_{x \to a}g(x)=b\implies\forall\varepsilon>0, \exists\delta_1:0<|x-a|<\delta_1\implies
    |g(x)-b|<\varepsilon$. $\lim_{t \to b}f(t)=L\implies\forall\varepsilon>0, \exists\delta_2:0<|t-
    b|<\delta_2\implies|g(x)-b|<\varepsilon$. $\forall\varepsilon>0$, choose $\delta_1$ such that
  $0<|x-a|<\delta_1\implies0<|g(x)-b|<\varepsilon=\delta_2$. By the second and third assumptions,
  such a $delta_1$ exists. In addition, $0<|g(x)-b|<\varepsilon=\delta_2\implies\left|f\left(
    g(x)\right)\right|<\varepsilon$. Therefore, $\lim_{x \to a}f\left(g(x)\right)=L$ by definition.
\end{proof}
