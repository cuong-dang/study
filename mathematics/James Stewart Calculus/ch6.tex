\section{Inverse Functions}
\subsection{Inverse Functions}
\begin{theorem*}
	If $f$ is a one-to-one differentiable function with inverse function $f^{-1}$ and $f'(f^{-1}(a))\ne0$,
	then the inverse function is differentiable at $a$ and
	\[(f^{-1})'(a)=\frac{1}{f'(f^{-1}(a))}.\]
\end{theorem*}

\begin{definition*}[The Number $e$]
	$e$ is the number such that $\lim_{h\to0}\frac{e^h-1}{h}=1$.
