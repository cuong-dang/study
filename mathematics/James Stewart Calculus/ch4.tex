\begin{definition*}
	The \textbf{area} $A$ of the region $S$ that lines under the graph
	of the continuous function $f$ is the limit of the sum of the areas
	of approximating rectangles:
	\[A=\lim_{n\to\infty}R_n=\lim_{n\to\infty}[f(x_1)\Delta
			x+\cdots+f(x_n)\Delta x].\]
\end{definition*}

\subsection{The Fundamental Theorem of Calculus}
\begin{theorem*}[The Fundamental Theorem of Calculus, Part 1]
	If $f$ is continuous on $[a, b]$, then the function $g$ defined by
	\[g(x)=\int_{a}^{x}f(t)dt \hspace{1em} a\le x\le b\] is continuous
	on $[a, b]$ and differentiable on $(a, b)$, and $g'(x)=f(x)$.
\end{theorem*}
