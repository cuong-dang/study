\subsection{Areas and Distances}
\begin{definition*}
  The \textbf{area} $A$ of the region $S$ that lines under the graph of the continuous function $f$
  is the limit of the sum of the areas of approximating rectangles:
  \[A=\lim_{n\to\infty}R_n=\lim_{n\to\infty}[f(x_1)\Delta x+\cdots+f(x_n)\Delta x].\]
\end{definition*}
