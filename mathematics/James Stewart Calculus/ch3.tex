\section{Applications of Differentiation}
\subsection{Maximum and Minimum Values}
\begin{definition*}
  Let $c$ be a number in the domain $D$ of a function $f$. Then $f(c)$ is the
  \begin{itemize}
    \item \textbf{absolute maximum} value of $f$ on $D$ if $f(c)\ge f(x)$ for all $x$ in $D$.
    \item \textbf{absolte minimum} value of $f$ on $D$ if $f(c)\le f(x)$ for all $x$ in $D$.
  \end{itemize}
\end{definition*}
