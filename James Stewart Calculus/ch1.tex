\subsection{Calculating Limits Using the Limit Laws}
\begin{enumerate}

      \item For all $x \ne 0$, $0 \le \sin^2(2\pi/2) \le 1$. Adding 1 to each side, we have $ 1 \le
                  1+\sin^2(2\pi/2) \le 2$. Because, $\sqrt{x} \ge 0$ for all $x \ge 0$, multiplying
            each side of the inequality gives $\sqrt{x} \le \sqrt{x}[1+\sin^2(2\pi/2)] \le
                  2\sqrt{x}$. It is true that $\lim_{x \to 0^+}{\sqrt{x}} = 0$ and $\lim_{x \to 0^+}
                  {2\sqrt{x}} = 0$. Applying the Squeeze Theorem proves that $\lim_{x \to 0^+}
                  {[1+\sin^2(2\pi/2)]} = 0$.
      \item $\lim_{x\to3^+}\left(2x+|x-3|\right) = \lim_{x\to3^+}(2x+x-3) = 6$. $\lim_{x\to3^-}
                  \left(2x+|x-3|\right) = \lim_{x\to3^-}(2x+3-x) = 6$. Thus, $\lim_{x\to3}
                  \left(2x+|x-3|\right) = 6$.

                  = \lim_{x\to0.5^-}\frac{1}{-x^2} = 4.$
