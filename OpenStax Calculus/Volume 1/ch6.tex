\section*{6.1 | Areas between Curves}
\begin{theorem*}[Finding the Area between Two Curves]
    Let $f(x)$ and $g(x)$ be continuous functions such that $f(x) \ge g(x)$ over an interval
    $\left[a, b\right]$. Let $R$ denote the region bounded above by the graph of $f(x)$, below by
    the graph of $g(x)$, and on the left and right by $x = a$ and $x = b$, respectively. Then, the
    area of $R$ is given by
    \[A = \int_{a}^{b}[f(x) - g(x)]dx.\]
\end{theorem*}
